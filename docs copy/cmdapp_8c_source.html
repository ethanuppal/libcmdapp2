43static bool ca_check_arg_consistency(int argc, const char* argv[]) {
44 // check: argc is strictly positive
45 if (argc < 1) {
46 return false;
47 }
48
49 // check: argv is non-NULL
50 if (!argv) {
51 return false;
52 }
53
54 // assume: `argv` is a valid pointer
55 // check: there are at least argc entries in argv
56 int i = 0;
57 while (i < argc) {
58 if (!argv[i]) {
59 return false;
60 }
61 i++;
62 }
63
64 return true;
65}

88/* Parses the given `behavior` string and initializes `opt`. Returns zero on
89 * success, nonzero otherwise. */
90static int ca_parse_opt_behavior(struct ca_opt* opt, const char* behavior) {
91 // no behavior, fall back to defaults
92 if (behavior[0] == 0) {
93 return 0;
94 }
95
96 size_t i = 0;
97
98 // otherwise we must specify that it takes an argument or is in multiflag
99 if (behavior[0] == '.') {
100 opt->flags |= CA_OPT_ARG;
101 i++;
102
103 if (isalpha(behavior[i])) {
104 opt->arg_name = behavior + 1;
105 while (isalpha(behavior[i])) {
106 i++;
107 }
108 }
109
110 // we check whether the argument is optional
111 if (behavior[i] == '?') {
112 opt->flags |= CA_OPT_OPTARG;
113 i++;
114 } else if (behavior[i] == '\0') {
115 return 0;
116 }
117
118 // check if no quantifier provided
119 if (behavior[i] == '\0') {
120 return 0;
121 }
122 } else if (behavior[0] == '*') {
123 opt->flags |= CA_OPT_MFLAG;
124 i++;
125
126 // check if no quantifier provided
127 if (behavior[i] == '\0') {
128 return 0;
129 }
130 }
131
132 // find the first position after . + ? + whitespace where we encounter a
133 // quantifier
134 while (strchr(".? \t", behavior[i]) != NULL
135 && strchr("!@&", behavior[i]) == NULL) {
136 i++;
137 }
138
139 // check whether quantifier is negated
140 if (behavior[i] == '!') {
141 opt->quantifier_is_negated = true;
142 i++;
143 }
144
145 // determine quantifier
146 switch (behavior[i]) {
147 case '@':
148 opt->quantifier = CA_OPT_QUANTIFIER_ANY;
149 break;
150 case '&':
151 opt->quantifier = CA_OPT_QUANTIFIER_ALL;
152 break;
153 case '<':
154 opt->quantifier = CA_OPT_QUANTIFIER_ONLY;
155 break;
156 default:
157 return 1;
158 }
159
160 // remaining input [i + 1, length) is list of flags
161 opt->refs = behavior + i + 1;
162
163 // validate that it is only a list of flags
164 for (size_t i = 0; opt->refs[i]; i++) {
165 if (!ca_is_short_flag(opt->refs[i])) {
166 return 1;
167 }
168 }
169
170 return 0;
171}

177static struct ca_opt* ca_lookup_opt(char short_opt, const char* long_opt) {
178 struct ca_opt* short_opt_cache[26 + 26 + 10] = {NULL};
179
180 if (short_opt != '\0') {
181 int map_index = islower(short_opt) ? short_opt - 'a'
182 : isupper(short_opt) ? short_opt - 'A' + 26
183 : isdigit(short_opt) ? short_opt - '0' + 26 + 26
184 : -1;
185 if (short_opt_cache[map_index]) {
186 return short_opt_cache[map_index];
187 }
188 for (size_t i = 0; i < app.options_length; i++) {
189 if (app.options[i].short_opt == short_opt) {
190 return short_opt_cache[map_index] = &app.options[i];
191 }
192 }
193 } else if (long_opt != NULL) {
194 for (size_t i = 0; i < app.options_length; i++) {
195 if (strcmp(app.options[i].long_opt, long_opt) == 0) {
196 return &app.options[i];
197 }
198 }
199 }
200 return NULL;
201}

226int ca_init(int argc, const char* argv[]) {
227 // ensure inputs are safe to use
228 if (!ca_check_arg_consistency(argc, argv)) {
229 errno = EINVAL;
230 return 1;
231 }
232 app.argc = argc;
233 app.argv = argv;
234
235 // the program name is in the first element of argv
236 app.program = argv[0];
237
238 // no default description
239 app.description = NULL;
240
241 // initialize empty authors array
242 app.authors = ca_dynamic_new(const char*, app.authors_length,
243 app.authors_capacity);
244 if (!app.authors) {
245 errno = ENOMEM;
246 return 1;
247 }
248
249 // no year provided initially
250 app.year = CA_NO_YEAR;
251
252 // v0.0.0
253 app.ver_major = 0;
254 app.ver_minor = 0;
255 app.ver_patch = 0;
256
257 // initialize empty synopses array
258 app.synopses = ca_dynamic_new(const char*, app.synopses_length,
259 app.synopses_capacity);
260 if (!app.synopses) {
261 errno = ENOMEM;
262 return 1;
263 }
264
265 // default additional versioning information
266 app.ver_info = "All rights reserved.";
267
268 // default: -- ends the option list
269 app.use_end_of_options = true;
270
271 // initialize empty synopses array
272 app.options = ca_dynamic_new(struct ca_opt, app.options_length,
273 app.options_capacity);
274 if (!app.options) {
275 errno = ENOMEM;
276 return 1;
277 }
278
279 // initialize empty results array
280 app.results = ca_dynamic_new(struct ca_parse_result, app.results_length,
281 app.results_capacity);
282 if (!app.results) {
283 errno = ENOMEM;
284 return 1;
285 }
286
287 // no callbacks by default
288 app.opt_callback = NULL;
289 app.arg_callback = NULL;
290
291 // supply default --help and --version implementations
292 app.override_help = false;
293 app.override_version = false;
294
295 // register deinitialization on program exit

354bool* ca_opt(char short_opt, const char* long_opt, const char* behavior,
355 const char** result, const char* description) {
356 // these parameters must be passed
357 if (!long_opt || !behavior) {
358 errno = EINVAL;
359 return NULL;
360 }
361 if (short_opt != '\0' && !ca_is_short_flag(short_opt)) {
362 errno = EINVAL;
363 return NULL;
364 }
365
366 // initialize option
367 struct ca_opt opt;
368 opt.short_opt = short_opt;
369 opt.long_opt = long_opt;
370 opt.result = result;
371 opt.flags = 0;
372 opt.quantifier_is_negated = false;
373 opt.quantifier = CA_OPT_QUANTIFIER_NONE;
374 opt.arg_name = "ARG";
375 opt.description = description;
376
377 // parse behavior
378 if (ca_parse_opt_behavior(&opt, behavior) != 0) {
379 errno = EINVAL;
380 return NULL;
381 }
382
383 // if it takes an arg, it has to store it somewhere
384 if (opt.flags & CA_OPT_ARG && !result) {
385 errno = EINVAL;
386 return NULL;
387 }
388
389 ca_dynamic_push(&app.options, app.options_length, app.options_capacity,
390 opt);
391
392 return &app.options[app.options_length - 1].was_passed;
393}

436static int ca_construct_results(void) {
437 // when -- is passed and support for it is enabled, all subsequent arguments
438 // are treated only as arguments
439 bool only_args_now = false;
440
441 // we start at 1 because argv[0] is already in app.program and it's useless
442 // here
443 int i = 1;
444
445 // the previous option that takes an argument
446 // this introduces code dup I need to figure out how to fix
447 // specifically, anytime there is arg_callback
448 struct ca_opt* last_opt = NULL;
449
450 // go through the command line arguments
451 while (i < app.argc) {
452 const char* cur = app.argv[i];
453
454 if (only_args_now) {
455 // handle the case of after --
456 ca_parsed_arg(&last_opt, cur);
457 i++;
458 } else if (cur[0] == '-') {
459 // it could be a flag
460
461 // handle '-' (common for stdin)
462 if (cur[1] == '\0') {
463 ca_parsed_arg(&last_opt, cur);
464 i++;
465 continue;
466 }
467
468 // handle '--'
469 if (strcmp(cur, "--") == 0) {
470 if (app.use_end_of_options) {
471 only_args_now = true;
472 } else {
473 ca_parsed_arg(&last_opt, cur);
474 }
475 i++;
476 continue;
477 }
478
479 // if last opt has not been resolved, we're at a flag now, so it
480 // better have needed an _optional_ argument
481 if (last_opt && !(last_opt->flags & CA_OPT_OPTARG)) {
482 ca_print_error("--%s missing required argument\n",
483 last_opt->long_opt);
484 return 1;
485 }
486 last_opt = NULL;
487
488 // we now parse the option and argument (if there)
489 struct ca_opt* opt = NULL;
490 const char* arg = NULL;
491
492 // determine whether it is a long or short option and search for the
493 // corresponding option struct
494 if (cur[1] != '-' /* short opt */) {
495 char flag = cur[1];
496
497 // the first character after '-' should always be a valid option
498 opt = ca_lookup_opt(flag, NULL);
499 if (!opt) {
500 ca_print_error("unknown flag -%c\n", flag);
501 return 1;
502 }
503
504 // we might have more characters though
505 if (cur[2] != '\0') {
506 // we have a bunch of flags potentially after a short opt
507 // can only be multiflag if first is multiflag and then all
508 // others are too
509 if (opt->flags & CA_OPT_MFLAG) {
510 for (size_t j = 2; cur[j]; j++) {
511 opt = ca_lookup_opt(cur[j], NULL);
512 if (!opt) {
513 ca_print_error("unknown flag -%c\n", cur[j]);
514 return 1;
515 }
516 if (!(opt->flags & CA_OPT_MFLAG)) {
517 ca_print_error(
518 "-%c must be passed separately from -%c\n",
519 opt->short_opt, flag);
520 return 1;
521 }
522 }
523 for (size_t j = 1; cur[j]; j++) {
524 // this will work now because we tested it above
525 opt = ca_lookup_opt(cur[j], NULL);
526 ca_parsed_opt(opt, NULL);
527 }
528 i++;
529 continue;
530 } else {
531 // treat as connected option
532 // example: -I/usr/include is -I /usr/include
533 opt = ca_lookup_opt(flag, NULL);
534 if (!(opt->flags & CA_OPT_ARG)) {
535 ca_print_error("-%c does not take arguments\n",
536 flag);
537 return 1;
538 }
539 arg = cur + 2;
540 }
541 }
542 } else /* long opt */ {
543 opt = ca_lookup_opt('\0', cur + 2);
544 if (!opt) {
545 ca_print_error("unknown flag %s\n", cur);
546 return 1;
547 }
548 }
549 if (!arg && opt->flags & CA_OPT_ARG) {
550 // delay resolution of argument until next iteration or loop end
551 last_opt = opt;
552 } else {
553 ca_parsed_opt(opt, arg);
554 }
555 i++;
556 } else {
557 // otherwise it must be an argument
558 ca_parsed_arg(&last_opt, cur);
559 i++;
560 }
561 }
562
563 // more code dup
564 if (last_opt && !(last_opt->flags & CA_OPT_OPTARG)) {
565 ca_print_error("--%s missing required argument\n", last_opt->long_opt);
566 return 1;
567 }
568
569 return 0;
570}

574static int ca_verify_results(void) {
575 // check every option passed
576 for (size_t i = 0; i < app.results_length; i++) {
577 struct ca_parse_result result = app.results[i];
578 if (result.opt) {
579 const struct ca_opt* opt = result.opt;
580 struct ca_opt* ref = NULL;
581
582 // determine if the quantified proposition holds
583 bool verdict = false;
584 switch (opt->quantifier) {
585 case CA_OPT_QUANTIFIER_ANY: {
586 // start out false, any success makes true
587 verdict = false;
588 for (size_t j = 0; opt->refs[j]; j++) {
589 ref = ca_lookup_opt(opt->refs[j], NULL);
590 if (!ref) {
591 ca_print_error(
592 "unknown flag -%c in definition of --%s\n",
593 opt->refs[j], opt->long_opt);
594 return 1;
595 }
596 if (ref->was_passed) {
597 verdict = true;
598 break;
599 }
600 }
601 break;
602 }
603 case CA_OPT_QUANTIFIER_ALL: {
604 // start out true, any failure makes false
605 verdict = true;
606 for (size_t j = 0; opt->refs[j]; j++) {
607 ref = ca_lookup_opt(opt->refs[j], NULL);
608 if (!ref) {
609 ca_print_error(
610 "unknown flag -%c in definition of --%s\n",
611 opt->refs[j], opt->long_opt);
612 return 1;
613 }
614 if (!ref->was_passed) {
615 verdict = false;
616 break;
617 }
618 }
619 break;
620 }
621 case CA_OPT_QUANTIFIER_ONLY: {
622 // 1. determine how many of the allowed arguments were
623 // passed
624 // 2. if more arguments were passed then allowed, there's a
625 // conflict
626 verdict = true;
627 size_t allowed_count = 0;
628 for (size_t j = 0; opt->refs[j]; j++) {
629 ref = ca_lookup_opt(opt->refs[j], NULL);
630 if (!ref) {
631 ca_print_error(
632 "unknown flag -%c in definition of --%s\n",
633 opt->refs[j], opt->long_opt);
634 return 1;
635 }
636 if (ref->was_passed) {
637 allowed_count++;
638 }
639 if (app.options_count > allowed_count) {
640 verdict = false;
641 }
642 }
643 break;
644 }
645 default: {
646 verdict = true;
647 break;
648 }
649 }
650 // negation flips verdict
651 if (opt->quantifier_is_negated) {
652 verdict = !verdict;
653 }
654 // render verdict
655 if (!verdict) {
656 switch (opt->quantifier) {
657 case CA_OPT_QUANTIFIER_ANY: {
658 if (opt->quantifier_is_negated) {
659 ca_print_error("-%c conflicts with --%s\n",
660 ref->short_opt, opt->long_opt);
661 } else {
662 ca_print_error(
663 "at least one of the specified options for "
664 "--%s must be passed\n",
665 opt->long_opt);
666 }
667 break;
668 }
669 case CA_OPT_QUANTIFIER_ALL: {
670 if (opt->quantifier_is_negated) {
671 ca_print_error(
672 "only some of the specified options for --%s "
673 "should be passed\n",
674 opt->long_opt);
675 } else {
676 ca_print_error(
677 "all of the specified options for --%s must be "
678 "passed\n",
679 opt->long_opt);
680 }
681 break;
682 }
683 case CA_OPT_QUANTIFIER_ONLY: {
684 if (opt->quantifier_is_negated) {
685 ca_print_error(
686 "only other options besides those specified "
687 "for --%s should be passed\n",
688 opt->long_opt);
689 } else {
690 if (opt->short_opt != '\0'
691 && opt->refs[0] == opt->short_opt) {
692 ca_print_error(
693 "--%s must be passed by itself\n",
694 opt->long_opt);
695 } else {
696 ca_print_error(
697 "--%s can only be passed with allowed "
698 "options\n",
699 opt->long_opt);
700 }
701 }
702 break;
703 }
704 default:
705 break;
706 }
707 return 1;
708 }
709 }
710 }
711
712 return 0;

715int ca_parse(void* user_data) {
716 // reset all options
717 for (size_t i = 0; i < app.options_length; i++) {
718 app.options[i].was_passed = false;
719 }
720 app.options_count = 0;
721 // clear results array
722 app.results_length = 0;
723
724 // do bulk of the parsing
725 if (ca_construct_results() != 0) {
726 return 1;
727 }
728
729 // check for conflicts
730 if (ca_verify_results() != 0) {
731 return 1;
732 }
733
734 // run the callbacks
735 for (size_t i = 0; i < app.results_length; i++) {
736 struct ca_parse_result result = app.results[i];
737 if (result.opt) {
738 if (!app.override_help
739 && strcmp(result.opt->long_opt, "help") == 0) {
740 ca_print_help();
741 } else if (!app.override_version
742 && strcmp(result.opt->long_opt, "version") == 0) {
743 ca_print_version();
744 } else {
745 app.opt_callback(result.opt->short_opt, result.opt->long_opt,
746 result.arg, user_data);
747 }
748 } else {
749 app.arg_callback(result.arg, user_data);
750 }
751 }
752
753 return 0;
754}

796void ca_print_help() {
797 // keep track of whether a section has been printed so extra space can be
798 // added for separation
799 bool previous_print = false;
800
801 // print description
802 if (app.description) {
803 printf("%s\n", app.description);
804 previous_print = true;
805 }
806
807 // print synopses
808 if (app.synopses_length > 0) {
809 if (previous_print) printf("\n");
810 printf("Usage: %s %s\n", app.program, app.synopses[0]);
811 for (size_t i = 1; i < app.synopses_length; i++) {
812 printf(" or: %s %s\n", app.program, app.synopses[i]);
813 }
814 previous_print = true;
815 }
816
817 // print options
818 if (app.options_length > 0) {
819 if (previous_print) printf("\n");
820 printf("Options:\n");
821 for (size_t i = 0; i < app.options_length; i++) {
822 const struct ca_opt* opt = &app.options[i];
823
824 // print the short option
825 if (opt->short_opt != '\0') {
826 printf(" -%c, ", opt->short_opt);
827 } else {
828 printf(" ");
829 }
830
831 // print the long option with arg if necessary
832 int chars_printed = printf("--%s", opt->long_opt);
833 if (opt->flags & CA_OPT_ARG) {
834 chars_printed += printf("[=");
835 for (size_t j = 0; isalpha(opt->arg_name[j]); j++) {
836 chars_printed += printf("%c", opt->arg_name[j]);
837 }
838 chars_printed += printf("]");
839 }
840
841 // determine if past the offset for printing descriptions
842 // if it is, put description on new line
843 // otherwise print description afterward
844 int padding = CA_DESCRIPTION_OFFSET
845 - (chars_printed + (sizeof(" ") - 1));
846 if (padding < 1 /* min of 1 prevents right next to each other */) {
